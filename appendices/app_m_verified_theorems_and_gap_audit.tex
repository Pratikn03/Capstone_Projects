\chapter{Verified Theorems and Gap Audit}
\label{app:verified-theorems-gap-audit}

This appendix mirrors the flagship T1--T8 battery theorem ladder used by the
proof appendix.  That ladder is the governance-critical proof spine, but it is
not the whole integrated theorem surface of the thesis; Appendix~\ref{app:integrated-theorem-surface-register}
audits the full theorem-like inventory.  The flagship set is:
\begin{enumerate}
  \item OASG existence,
  \item one-step safety preservation,
  \item ORIUS core bound,
  \item no free safety,
  \item certificate validity horizon,
  \item certificate expiration bound,
  \item feasible fallback existence, and
  \item graceful degradation dominance.
\end{enumerate}
The certificate half-life remains a corollary, and safe-budget monotonicity
remains a proposition.

\section{Theorem Statements}

\subsection{Theorem 1: OASG Existence}
\label{sec:thm-oasg}

\textbf{Statement.} There exist scenarios where an action appears safe in observed state but is unsafe in true physical state. Formally: $\exists$ episodes with degraded telemetry such that $\text{observed-safe}(a \mid x_{\text{obs}}) \neq \text{true-safe}(a \mid x_{\text{true}})$.

\textbf{Assumptions.} A2 (bounded telemetry error), A4 (known dynamics). Fault injection produces non-trivial $x_{\text{obs}} \neq x_{\text{true}}$ separation.

\textbf{Proof status.} Structural: code explicitly separates observed and true state; empirical: violation rates under dropout/drift exceed nominal.

\textbf{Code anchor.} \texttt{src/orius/cpsbench\_iot/}: \texttt{scenarios.py} (fault generation, $x_{\text{obs}}$, $x_{\text{true}}$, \texttt{event\_log}); \texttt{plant.py} (unclamped truth SOC in \texttt{BatteryPlant.step()}); \texttt{runner.py} (true-state violation computation).

\subsection{Theorem 2: One-Step Safety Preservation}
\label{sec:thm-safety-preservation}

\textbf{Statement.} If the repaired action lies inside the tightened safe set
and the current state lies inside the inflated certificate, then the next
battery state remains safe.

\textbf{Assumptions.} A1--A5, A7.

\textbf{Proof status.} Formal chapter proof plus deterministic runtime
guarantee checks.

\textbf{Code anchor.} \texttt{src/orius/dc3s/guarantee\_checks.py} and
\texttt{src/orius/dc3s/shield.py}.

\subsection{Theorem 3: ORIUS Core Bound}
\label{sec:thm-core-bound}

\textbf{Statement.} The expected battery violation count satisfies
$\mathbb{E}[V] \le \alpha(1-\bar{w})T$.

\textbf{Assumptions.} A1--A5, A7.

\textbf{Proof status.} Formal chapter proof plus explicit runtime helpers for
computing and checking the empirical bound.

\textbf{Code anchor.} \texttt{src/orius/dc3s/coverage\_theorem.py}.

\subsection{Theorem 4: No Free Safety}
\label{sec:thm-no-free}

\textbf{Statement.} Any controller maintaining zero true-state violations must accept at least some intervention rate. There is no controller that achieves both zero violations and zero interventions under admissible fault sequences.

\textbf{Assumptions.} A3 (feasible safe repair exists), A8 (admissible fallback policy). Quality-ignorant baselines lack reliability-aware inflation.

\textbf{Proof status.} Conceptual: follows from T1 (OASG existence) and the necessity of repair when observed-safe $\neq$ true-safe. Empirical: DC3S achieves zero violation with non-zero intervention; baselines achieve non-zero violation.

\textbf{Code anchor.} Supporting evidence in \texttt{baselines.py},
\texttt{scenarios.py}, and \texttt{runner.py}.

\subsection{Theorem 5: Certificate Validity Horizon}
\label{sec:thm-cert-horizon}

\textbf{Statement.} A battery certificate remains valid for the largest
integer horizon whose forward tube remains inside the battery SOC bounds.

\textbf{Assumptions.} A1, A2, A4--A7.

\textbf{Proof status.} Formal chapter theorem with direct validation helper.

\textbf{Code anchor.} \texttt{src/orius/dc3s/temporal\_theorems.py}
(\texttt{forward\_tube()}, \texttt{certificate\_validity\_horizon()}).

\subsection{Theorem 6: Certificate Expiration Bound}
\label{sec:thm-cert-expiration}

\textbf{Statement.} The battery certificate horizon is lower-bounded by the
issuance-time distance to the nearest SOC boundary divided by the drift scale.

\textbf{Assumptions.} A4--A7.

\textbf{Proof status.} Formal chapter theorem with exact code implementation
of the analytical lower bound.

\textbf{Code anchor.} \texttt{src/orius/dc3s/temporal\_theorems.py}
(\texttt{certificate\_expiration\_bound()}).

\subsection{Theorem 7: Feasible Fallback Existence}
\label{sec:thm-fallback}

\textbf{Statement.} There exists a battery fallback action that preserves
safety from an interior SOC state under bounded model error.

\textbf{Assumptions.} A3, A4, A8.

\textbf{Proof status.} Constructive battery fallback theorem.

\textbf{Code anchor.} \texttt{src/orius/dc3s/temporal\_theorems.py}
(\texttt{zero\_dispatch\_fallback()}, \texttt{certify\_fallback\_existence()}).

\subsection{Theorem 8: Graceful Degradation Dominance}
\label{sec:thm-graceful-dominance}

\textbf{Statement.} The battery graceful-degradation controller incurs no more
violations than the uncontrolled controller under the same admissible fault
sequence, with strict improvement whenever the uncontrolled path violates.

\textbf{Assumptions.} A1--A5, A8.

\textbf{Proof status.} Formal chapter theorem with theorem-specific comparison
helper and safe-landing implementation support.

\textbf{Code anchor.} \texttt{src/orius/dc3s/temporal\_theorems.py}
(\texttt{evaluate\_graceful\_degradation\_dominance()}) and
\texttt{src/orius/dc3s/shield.py} (\texttt{safe\_landing} mode).
