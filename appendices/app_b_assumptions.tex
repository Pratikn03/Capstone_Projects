\chapter{Assumption Register}
\label{app:assumptions}

This appendix is the canonical assumption register for the defended ORIUS
theorem program.  The active theorem registry in
\texttt{reports/publication/theorem\_registry.yml} references these IDs
directly, and the audit scripts reject theorem rows that cite an unknown
assumption label.

\section{How to Read This Register}

Each assumption is stated as a theorem-facing predicate and then grounded in
the battery runtime.  The defended headline core is
\[
  \{T1,\;T2,\;T3a,\;T4,\;T11,\;T_{\mathrm{trajectory\_PAC}}\},
\]
with $T6$ retained as a formal auxiliary theorem.  Draft or scoped extension
surfaces may cite these assumptions too, but they do not enter defended-core
counts unless the registry says so.

\section{Detailed Statements}

\begin{assumption}[A1 --- Almost-sure model error bound]\label{assum:A1}
For all $t$,
\[
  |x_{t+1} - f(x_t,a_t)| \le \epsilon_{\mathrm{model}}
\]
almost surely.  This is stronger than the sub-Gaussian tail control in
Assumption~A9 and is used when the proof requires a deterministic one-step
postcondition rather than a high-probability statement.
\textbf{Battery interpretation:} the one-step battery model error is absorbed
as an explicit MWh slack in the FTIT/shield tightening margin.
\end{assumption}

\begin{assumption}[A2 --- Telemetry-state bridge]\label{assum:A2}
There exists a known monotone function $g\colon[0,1]\to\mathbb{R}_+$ such that
\[
  |\hat{x}_t - x_t| \le g(w_t), \qquad g(1)=0,
\]
for all $t$.
\textbf{Battery interpretation:} the OQE reliability score $w_t$ controls the
worst-case telemetry-state discrepancy used by the tightened SOC tube.
\end{assumption}

\begin{assumption}[A3 --- Feasible safe repair]\label{assum:A3}
Whenever the runtime invokes repair, the tightened safe-action set is non-empty.
\textbf{Battery interpretation:} the battery shield projection has at least one
admissible charge/discharge action inside the FTIT-restricted SOC region.
\end{assumption}

\begin{assumption}[A4 --- Known one-step dynamics]\label{assum:A4}
The one-step increment map $\Delta(a)=f(x,a)-x$ is known exactly to the
controller.
\textbf{Battery interpretation:} the SOC update induced by charge, discharge,
step duration, and efficiencies is computed explicitly by the runtime.
\end{assumption}

\begin{assumption}[A5 --- Absorbed monotone tightening]\label{assum:A5}
Let $m_t$ denote the reliability-conditioned tightening margin.  The defended
T2/T3 surface uses the absorbed margin
\[
  m_t^\star = m_t + \epsilon_{\mathrm{model}},
\]
and assumes $m_t^\star$ is monotone non-increasing in $w_t$ and bounded above
by a finite constant.
\textbf{Battery interpretation:} the FTIT margin combines the base
reliability/certificate slack with the one-step model-error allowance before
the shield constructs the admissible SOC region.
\end{assumption}

\begin{assumption}[A6 --- Bounded detector lag]\label{assum:A6}
Each supported telemetry fault class is detected within a finite lag
$\tau_{\max}$ known to the runtime.
\textbf{Battery interpretation:} dropout, stale, delay, out-of-order, spike,
and drift channels each have a finite audit-locked response horizon in the
runtime configuration.
\end{assumption}

\begin{assumption}[A7 --- Causal certificate rule]\label{assum:A7}
The certificate state obeys
\[
  \mathrm{cert}_t = h(z_1,\ldots,z_t),
\]
for a causal update rule $h$.
\textbf{Battery interpretation:} certificate issuance, renewal, and runtime
bookkeeping never use future telemetry.
\end{assumption}

\begin{assumption}[A8 --- Safe-interior fallback]\label{assum:A8}
There exists a fallback controller $\pi_{\mathrm{fb}}$ such that
\[
  \TSVR(\pi_{\mathrm{fb}})=0
\]
whenever the latent state is in the safe interior with at least
$\epsilon_{\mathrm{model}}$ slack at the moment the fault is handled.
\textbf{Battery interpretation:} zero-dispatch or equivalent safe holding
behavior is only claimed from an interior SOC state with explicit room for the
one-step model error.
\end{assumption}

\begin{assumption}[A9 --- Sub-Gaussian disturbance law]\label{assum:A9}
The disturbance increments satisfy
\[
  \mathbb{E}\!\left[e^{s\epsilon_t}\right] \le e^{s^2\sigma_d^2/2}
  \quad \text{for all } s\in\mathbb{R}.
\]
\textbf{Battery interpretation:} calibrated residuals are modeled with a known
sub-Gaussian proxy parameter $\sigma_d$, which drives the defended T6 bound and
the trajectory-PAC helper.
\end{assumption}

\begin{assumption}[A10a --- Polynomial mixing telemetry]\label{assum:A10a}
The telemetry process is $\varphi$-mixing with polynomial decay
\[
  \varphi(k)=O(k^{-\beta})
  \quad \text{for some } \beta>1.
\]
\textbf{Battery interpretation:} this is the weaker mixing regime used only for
scoped weak-dependence arguments where polynomial decay is sufficient.
\end{assumption}

\begin{assumption}[A10b --- Geometric mixing telemetry]\label{assum:A10b}
The telemetry process is $\varphi$-mixing with geometric decay,
\[
  \varphi(k)\le C\rho^k
  \quad \text{for some } C>0,\; \rho\in(0,1).
\]
\textbf{Battery interpretation:} this is the stronger dependence-decay
assumption required by the current T9 separated-window proof.
\end{assumption}

\begin{assumption}[A11 --- Arbitrage boundary reachability]\label{assum:A11}
Under the admissible-demand family, the evaluated controller can be driven to a
boundary-sensitive witness state where the safety margin is tight enough for an
OASG/no-free-safety construction.
\textbf{Battery interpretation:} the witness controller can be steered near the
relevant SOC boundary without violating the admissible-demand model.
\end{assumption}

\begin{assumption}[A12 --- Controller-fault independence]\label{assum:A12}
Conditioned on the available history, the fault/reliability process is
independent of the controller's action stream.
\textbf{Battery interpretation:} telemetry degradation is modeled as exogenous
relative to the dispatch choice, once the observed history is fixed.
\end{assumption}

\begin{assumption}[A13 --- TV bridge]\label{assum:A13}
For the T10 binary boundary-testing observation laws,
\[
  \mathrm{TV}(P_{0,t}, P_{1,t}) \le w_t
\]
for every $t$.
\textbf{Battery interpretation:} the reliability score is treated as an
explicit upper bound on total-variation distinguishability for that scoped
binary lower-bound construction.
\end{assumption}
