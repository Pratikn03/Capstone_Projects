\subsection{Deep Theoretical Guarantees}\label{sec:deep_guarantees}

The earlier defended theorems (T1--T8) establish the runtime composition:
reliability-dependent inflation, monotone safety margins under the designed
inflation rule, and certificate soundness.  This longform section records draft
quantitative audit lemmas.  They are useful for internal review and theorem
development, but they are not presented as new conditional conformal-coverage
theorems without the stated binning, exchangeability, and audit-sample
assumptions.

\subsubsection{Draft T9: Reliability-Binned Finite-Sample Coverage Audit}

\begin{theorem}[Finite-sample grouped coverage audit]\label{thm:deep-finite-sample}
Fix a reliability bin $B$ and an audit split whose examples are exchangeable
within that bin.  Let
$Z_i=\mathbf{1}\{Y_i\in \hat C_i(X_i)\}$ for the $n_B$ audited examples in
$B$, and let $\widehat p_B=n_B^{-1}\sum_i Z_i$.  Then, for any
$\delta>0$:
\[
  \Pr\left(
    p_B \ge \widehat p_B -
    \sqrt{\frac{\ln(1/\delta)}{2\,n_B}}
  \right) \ge 1-\delta,
\]
where $p_B=\mathbb{E}[Z\mid w\in B]$ is the binwise coverage probability.  For
a weighted audit, the same displayed form is used with the Kish effective
sample size $n_{\mathrm{eff}}=(\sum_i w_i)^2/\sum_i w_i^2$ replacing $n_B$.
\end{theorem}

\begin{proof}[Proof sketch]
Within a fixed reliability bin, the audit indicators $Z_i$ are bounded in
$[0,1]$.  Applying Hoeffding's inequality to the binwise empirical mean gives
\[
  \Pr\bigl(p_B < \widehat p_B-\epsilon\bigr)
  \le \exp(-2n_B\epsilon^2).
\]
Solving for $\epsilon$ at confidence $\delta$ gives the displayed lower
confidence bound.  The weighted-audit variant substitutes the Kish effective
sample size as a conservative summary of weighted audit mass; it is an audit
bound, not a proof that inverse-reliability weighting preserves conformal
exchangeability.
\end{proof}

\paragraph{Implications.}
This result gives a computable lower confidence bound on measured grouped PICP
before deployment.  It should be reported beside the observed binwise coverage,
the bin count, and the effective sample size.  The executable witness is
\texttt{dc3s/theoretical\_guarantees.py:
compute\_finite\_sample\_coverage\_bound()}.

\subsubsection{Draft T10: Boundary-Indistinguishability OASG Lower Bound}

\begin{theorem}[Boundary-indistinguishability OASG lower bound]\label{thm:deep-separation}
Fix a horizon $T$.  For each step $t$, suppose there is a boundary-testing
subproblem with latent hypotheses $H_{0,t}$ (safe side) and $H_{1,t}$ (unsafe
side).  Let the observation laws under those hypotheses be $P_{0,t}$ and
$P_{1,t}$, and assume:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item the controller acts only through the degraded observation channel;
  \item $\mathrm{TV}(P_{0,t},P_{1,t})\le w_t$;
  \item $\Pr(H_{1,t})\ge p_t$ for a supplied boundary-mass sequence $p_t$;
  \item an unsafe-side mistake has OASG severity at least $\delta_t$.
\end{enumerate}
Then every observation-only controller $\pi$ satisfies
\[
  \mathbb{E}[V_T(\pi)]
  \;\ge\;
  \frac{1}{2}\sum_{t=1}^T p_t(1-w_t),
\]
and the severity-weighted OASG obeys
\[
  \mathbb{E}[\mathrm{OASG}_T(\pi)]
  \;\ge\;
  \frac{1}{2}\sum_{t=1}^T \delta_t p_t(1-w_t).
\]
\end{theorem}

\begin{proof}[Proof sketch]
Fix $t$.  Any observation-only release rule induces a binary test between the
safe-side and unsafe-side observation laws.  Le~Cam's two-point lemma gives
\[
  \max\{\Pr(\psi_t\ne 0\mid H_{0,t}),\Pr(\psi_t\ne 1\mid H_{1,t})\}
  \ge \frac{1-\mathrm{TV}(P_{0,t},P_{1,t})}{2}
  \ge \frac{1-w_t}{2}.
\]
On the unsafe side, an error releases an action that is unsafe for the true
state.  Multiplying by the unsafe-side boundary mass $p_t$ gives the per-step
violation-count lower bound.  Multiplying the same event by the supplied
minimum severity $\delta_t$ gives the per-step OASG lower bound.  Summing over
$t$ and using linearity of expectation proves both displays.
\end{proof}

\paragraph{Implications.}
T10 gives the rate that the qualitative no-free-safety lemma does not provide:
OASG exposure grows linearly in the observation degradation budget
$\sum_t(1-w_t)$, scaled by boundary mass and violation severity.  It is strong
when those quantities are measured or bounded in a domain, and it remains
honest because the boundary-mass and total-variation hypotheses are explicit.
The executable witness is
\texttt{dc3s/theoretical\_guarantees.py:
compute\_stylized\_frontier\_lower\_bound()}.

\subsubsection{Theorem T11: Adaptive Inflation Regret Bound}

\begin{theorem}[Adaptive Inflation Regret]\label{thm:deep-regret}
Let $I_t^*$ be the oracle-optimal inflation factor at step $t$ (known only in
hindsight) and $I_t$ the DC\textsuperscript{3}S inflation factor under
exponential smoothing with time constant $\tau$ and rate
$\gamma = 1 - e^{-1/\tau}$.  The cumulative squared tracking error satisfies:
\[
  R_T \;=\; \sum_{t=1}^T (I_t - I_t^*)^2
  \;\le\; \tau \cdot \Delta_{\max}^2 \cdot (1 + \ln T)
         \;+\; \frac{\Delta_{\max}^2}{\gamma^2}
\]
where $\Delta_{\max} = \max_t |I_t^* - I_{t-1}^*|$.  The per-step amortised
regret is $R_T / T = O(\tau \cdot \Delta_{\max}^2 \cdot \ln T / T) \to 0$
as $T \to \infty$.
\end{theorem}

\begin{proof}[Proof sketch]
The DC\textsuperscript{3}S inflation update is $I_t = (1 - \gamma)\,I_{t-1} + \gamma\,I_t^{\mathrm{target}}$.
This is a first-order IIR filter with step response bounded by
$(1 - \gamma)^k \cdot \Delta_{\max}$ after $k$ steps from a jump.  The
tracking error at each step is therefore bounded by the filter's transient
response.  Squaring and summing over $T$ steps with the geometric decay of
$(1 - \gamma)^k$ gives the $\tau \cdot \ln T$ tracking term.  The transient
term $\Delta_{\max}^2 / \gamma^2$ accounts for the worst single jump.
\end{proof}

\paragraph{Implications.}
The sublinear ($O(\ln T / T)$) per-step regret confirms that
DC\textsuperscript{3}S with time constant $\tau = 30$ (the locked production
value) tracks the oracle with diminishing per-step error.  For $T = 500$,
$\tau = 30$, and $\Delta_{\max} = 0.3$ (typical for moderate degradation
events), the bound gives $R_T / T \le 0.082$.
The executable witness is \texttt{dc3s/theoretical\_guarantees.py:
compute\_adaptive\_regret\_bound()}.
