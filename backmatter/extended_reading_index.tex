\section*{Suggested Extended Reading}

\begin{description}[style=nextline]

\item[Universal Safety Foundations]
  Angelopoulos and Bates, \emph{Conformal Prediction: A Gentle Introduction} (2023);
  Vovk, Gammerman, and Shafer, \emph{Algorithmic Learning in a Random World} (2005);
  Shafer and Vovk, ``A Tutorial on Conformal Prediction'' (JMLR 2008).

\item[Conformal Prediction and Calibration]
  Romano, Patterson, and Cand\`{e}s, ``Conformalized Quantile Regression'' (NeurIPS 2019);
  Gibbs and Cand\`{e}s, ``Adaptive Conformal Inference Under Distribution Shift'' (NeurIPS 2021);
  Barber et al., ``Conformal Prediction Beyond Exchangeability'' (AoS 2023);
  Tibshirani et al., ``Conformal Prediction Under Covariate Shift'' (2019);
  Zaffran et al., ``Adaptive Conformal Predictions for Time Series'' (2022).

\item[Runtime Assurance and Safe Control]
  Ames et al., ``Control Barrier Functions: Theory and Applications'' (2019);
  Bloem et al., ``Shield Synthesis'' (TACAS 2015);
  Wabersich and Zeilinger, ``A Predictive Safety Filter for Learning-Based Control'' (2021);
  Alshiekh et al., ``Safe Reinforcement Learning via Shielding'' (2018);
  Desai et al., ``SOTER'' (IROS 2019).

\item[Degraded Observation and CPS Trust]
  Chandola, Banerjee, and Kumar, ``Anomaly Detection: A Survey'' (2009);
  Liu, Ting, and Zhou, ``Isolation Forest'' (2008);
  Teixeira et al., ``A Secure Control Framework for Resource-Limited Adversaries'' (2015);
  Cardenas et al., ``Attacks Against Process Control Systems'' (2011);
  Giraldo et al., ``A Survey of Physics-Based Attack Detection in Cyber-Physical Systems'' (2018).

\item[Domain Instantiations]
  Energy and storage: Tarascon and Armand (2001), Dunn et al. (2011), Rosewater et al. (2019), Hossain et al. (2020).
  Automated driving and robotics: Koopman and Wagner (2016), Shalev-Shwartz et al. (2017), Althoff and Dolan (2014), Herbert et al. (2017).
  Industrial systems: Qin and Badgwell (2003), Yin et al. (2014), Isermann (2005), Zhang and Jiang (2008).
  Healthcare CPS: Lee and Sokolsky (2010), Saeed et al. (2011), Clifford et al. (2009), FDA SaMD guidance (2019).
  Navigation and planning: Kaelbling, Littman, and Cassandra (1998), Julier and Uhlmann (2004), Rockafellar and Uryasev (2000).
  Aerospace assurance: RTCA DO-178C (2011), Rushby (1993), Habli and Kelly (2010), Srinivasan and Sanner (2020).

\item[Reproducibility and Scientific Software]
  Gneiting and Raftery (2007); Makridakis et al. (2020); Lei et al. (2018); Barber, Cand\`{e}s, Ramdas, and Tibshirani (2023).
  Use the locked artifact index and publication manifests for all thesis-grade claims.

\end{description}
