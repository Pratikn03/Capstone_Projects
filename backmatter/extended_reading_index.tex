\section*{Suggested Extended Reading}

\begin{description}[style=nextline]

\item[Conformal Prediction]
  Vovk, Gammerman, and Shafer, \emph{Algorithmic Learning in a Random World} (2005);
  Romano, Patterson, and Cand\`{e}s, ``Conformalized Quantile Regression'' (NeurIPS 2019);
  Gibbs and Cand\`{e}s, ``Adaptive Conformal Inference Under Distribution Shift'' (NeurIPS 2021).

\item[Runtime Safety and Shields]
  Bloem et al., ``Shield Synthesis'' (TACAS 2015);
  Wachi et al., ``Safe Reinforcement Learning in Constrained MDPs'' (ICML 2024);
  Ames et al., ``Control Barrier Functions'' (IEEE TAC 2017).

\item[Battery Energy Storage Dispatch]
  Weitzel and Glock, ``Energy storage dispatch in energy markets'' (Energy 2018);
  Xu et al., ``Optimal battery storage operation'' (IEEE Trans.\ Smart Grid 2020).

\item[Cyber-Physical Systems Security]
  Cardenas et al., ``Attacks Against Process Control Systems'' (ASIACCS 2011);
  Teixeira et al., ``Secure Estimation and Control for CPS'' (ARCS 2015).

\item[Time-Series Forecasting]
  Lim and Zohren, ``Time-series forecasting with deep learning: a survey'' (Phil.\ Trans.\ Roy.\ Soc.\ A, 2021).

\end{description}
